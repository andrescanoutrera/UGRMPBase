242void Profile::append(const KmerFreq& kmerFreq) {
243 int pos = this->findKmer(kmerFreq.getKmer().toString());
244 if (pos >= 0) { // If found
245 this->at(pos).setFrequency(this->at(pos).getFrequency() +
246 kmerFreq.getFrequency());
247 } else { // If not found
248 if(this->_size == this->_capacity){ // If the vector is full
249 this->reallocate(this->_capacity + BLOCK_SIZE);
250 }
251 this->_size++;
252 this->at(this->_size-1) = kmerFreq;
253 }
